# 비밀번호 변경 후 기존 세션이 유지되는 보안 이슈

## 비밀번호 변경 후 기존 세션 무효화 구현

### 1단계: 비밀번호 변경 시 세션 버전 업데이트
- 사용자 데이터베이스에 `session_version` 컬럼 추가
- 비밀번호 변경 시 `session_version` 증가

javascript
const updatePassword = async (userId, newPassword) => {
  const hashedPassword = await bcrypt.hash(newPassword, 10);
  await db.user.update(
    { id: userId },
    { password: hashedPassword, session_version: { increment: 1 } }
  );
};


### 2단계: JWT 토큰에 세션 버전 포함
- 토큰 발급 시 현재 `session_version` 을 클레임에 추가

javascript
const generateToken = async (userId) => {
  const user = await db.user.findUnique({ where: { id: userId } });
  const token = jwt.sign(
    { userId, sessionVersion: user.session_version },
    process.env.JWT_SECRET,
    { expiresIn: '24h' }
  );
  return token;
};


### 3단계: 토큰 검증 시 세션 버전 확인
- 미들웨어에서 토큰의 `session_version`과 데이터베이스 값 비교

javascript
const verifyToken = async (req, res, next) => {
  try {
    const token = req.headers.authorization?.split(' ')[1];
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    const user = await db.user.findUnique({ where: { id: decoded.userId } });
    
    if (decoded.sessionVersion !== user.session_version) {
      return res.status(401).json({ error: 'Session invalidated' });
    }
    
    req.userId = decoded.userId;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
};


### 4단계: 로그아웃 시 모든 세션 무효화 (선택사항)
javascript
const logoutAllDevices = async (userId) => {
  await db.user.update(
    { id: userId },
    { session_version: { increment: 1 } }
  );
};
